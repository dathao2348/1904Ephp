Mệnh đề WHERE
        được dùng để lọc kết quả từ câu truy vấn
        Cú pháp:
        SELECT cột 1, cột 2...
        FROM tên bảng
        WHERE điều kiện;

        Ví dụ 1:
        láy ra tất cả các bản ghi có id=5
        SELECT * FROM customers
        WHERE id=5

        Ví dụ 2:
        Lấy ra tất cả các khách hàng đến từ Chicago
        SELECT * FROM customers
        WHERE city="Chicago"

        Lưu ý là số như id thì không cần dấu nháy bao nhưng chuỗi thì cần nháy đơn hoặc nháy kép để bao

        Ví dụ 3:
        Lấy ra tất cả các khách hàng có id lớn hơn 5
        SELECT * FROM customers
        WHERE id>5

        Ví dụ 4: Lọc theo điều kiện

        AND(đáp ứng tất cả các điều kiện)
        OR( chỉ cần đáp ứng 1 trong các điều kiện)

        Mệnh để OR:
        SELECT * FROM customers
        WHERE city="Chiacago" OR city="Los Angelas"

        Mệnh đề AND:
        Lấy ra tất cả các khách hàng có nghề nghiệp là Purchasing Manager và đến từ Chicago

        SELECT id, job_title,last_name,first_name,city FROM customers WHERE job_title="Purcharsing Manager" AND city="Chicago"

        Ví dụ 6: từ khóa NOT có nghĩa là phủ định
        Lấy ra tất cả các khách hàng mà không đến từ Chicago
        SELECT * FROM customers WHERE NOT city="Chicago"
